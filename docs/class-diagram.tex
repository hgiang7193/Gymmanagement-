\section{Thiết kế sơ đồ lớp}

% ------------------------------------------------------------
\subsection{Kiến trúc phân tầng}
% ------------------------------------------------------------

Hệ thống MYFIT được tổ chức theo kiến trúc \textbf{4 tầng phân lớp},
trong đó mỗi tầng chỉ phụ thuộc xuống tầng phía dưới và không bao giờ
phụ thuộc ngược lên:

\begin{table}[h]
\centering
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|>{\centering\arraybackslash}m{3cm}|>{\raggedright\arraybackslash}m{8cm}|}
\hline
\textbf{STT} & \textbf{Tầng} & \textbf{Trách nhiệm} \\ \hline
1 & Trình bày \newline \texttt{<<controller>>} &
    Nhận yêu cầu HTTP, xác thực đầu vào, uỷ quyền cho tầng Nghiệp vụ,
    trả về phản hồi HTTP. Không chứa logic nghiệp vụ. \\ \hline
2 & Nghiệp vụ \newline \texttt{<<service>>} &
    Triển khai toàn bộ luồng nghiệp vụ từ các ca sử dụng. Mỗi
    lớp Service tương ứng với một nhóm UC cùng miền. \\ \hline
3 & Truy cập dữ liệu \newline \texttt{<<repository>>} &
    Trừu tượng hoá truy vấn cơ sở dữ liệu qua giao diện.
    Tầng Nghiệp vụ không trực tiếp viết SQL, chỉ gọi phương thức Repository. \\ \hline
4 & Miền \newline \texttt{<<entity>>} &
    Ánh xạ 1-1 với bảng cơ sở dữ liệu. Chứa trường dữ liệu
    và các ràng buộc cơ bản. \\ \hline
\end{tabular}
\caption{Bốn tầng kiến trúc của hệ thống MYFIT}
\end{table}

\noindent Các nguyên tắc thiết kế xuyên suốt:

\begin{enumerate}
    \item \textbf{1 UC = 1 phương thức Service} — mỗi ca sử dụng ánh xạ
          thành ít nhất 1 phương thức trong lớp Service tương ứng.
    \item \textbf{Chốt giá tại thời điểm giao dịch} —
          \texttt{InvoiceLineItem} lưu \texttt{price\_snapshot} thay vì
          khoá ngoại động để tránh sai lệch khi gói bị sửa giá sau.
    \item \textbf{Đích đánh giá đa hình} —
          \texttt{Review.target\_type + target\_id} thay cho 4 bảng nối
          riêng (ca tập / huấn luyện viên / thiết bị / bài tập).
    \item \textbf{Nhật ký kiểm toán tập trung} — mọi sự kiện nghiệp vụ ghi vào
          \texttt{AuditLog} duy nhất thông qua tầng Service; tầng Controller
          không ghi trực tiếp.
\end{enumerate}

\subsection{Lớp thực thể (Entity Layer)}
% ------------------------------------------------------------

Lớp thực thể gồm \textbf{10 gói} và \textbf{22 lớp}, mỗi lớp ánh
xạ 1-1 với một bảng trong cơ sở dữ liệu. Toàn bộ được trình bày qua
3 sơ đồ theo nhóm chức năng.

\textbf{RefreshSession} &
  token\_hash, expires\_at, revoked &
  Lưu mã làm mới đã băm. Toàn bộ bị thu hồi khi đăng xuất
  hoặc đặt lại mật khẩu (UC-AUTH-02, UC-AUTH-03). \\ \hline

\textbf{Password Reset Token} &
  token\_hash, expires\_at, used &
  Mã dùng một lần, thời hạn 24 giờ. Dùng cho cả ``Quên mật khẩu''
  (UC-AUTH-03) và ``Đặt mật khẩu lần đầu'' (UC-ADMIN-02). \\ \hline

\textbf{Branch} &
  name, address, open/close\_time, zalo\_link, status: BranchStatus &
  Đơn vị tổ chức cơ bản. Mọi dữ liệu vận hành đều thuộc phạm vi
  theo \texttt{branch\_id}. \\ \hline

\textbf{Branch Manager Assignment} &
  manager\_id, branch\_id, assigned\_at &
  Bảng trung gian nhiều-nhiều: 1 Quản lý có thể phụ trách nhiều chi
  nhánh; Quản trị viên gán qua UC-ADMIN-02. \\ \hline

\textbf{AuditLog} &
  event\_type, entity\_type, entity\_id, payload: JSON &
  Chỉ ghi thêm, không sửa xoá. Ghi lại mọi sự kiện nghiệp vụ.
  \texttt{actor\_id} cho phép null để hỗ trợ sự kiện do hệ thống tự kích hoạt. \\ \hline

\subsection{Lớp truy cập dữ liệu (Repository Layer)}
% ------------------------------------------------------------

Tầng Repository cung cấp giao diện để tầng Nghiệp vụ truy vấn
cơ sở dữ liệu mà không biết chi tiết câu lệnh SQL. Mỗi giao diện bao gồm
các phương thức thao tác dữ liệu cơ bản và các truy vấn nghiệp vụ đặc thù.

\begingroup
\renewcommand{\arraystretch}{1.3}
\begin{longtable}{|>{\centering\arraybackslash}m{3.8cm}|>{\ttfamily\raggedright\arraybackslash}m{10.2cm}|}
\hline
\textbf{Giao diện} & \normalfont\textbf{Phương thức nghiệp vụ đặc thù} \\ \hline
