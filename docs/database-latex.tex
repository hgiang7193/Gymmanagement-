\multicolumn{2}{|l|}{\textbf{Nhóm 1: Người dùng và phân quyền}} \\ \hline

\textbf{users} \newline (người\_dùng) &
\begin{itemize}
    \item \texttt{id} (text, khóa chính).
    \item \texttt{email} (text, duy nhất): Email đăng nhập.
    \item \texttt{password\_hash} (text): Mật khẩu đã băm bằng Argon2id.
    \item \texttt{status} (text): Trạng thái tài khoản — \texttt{'active'}, \texttt{'inactive'}, \texttt{'suspended'}, \texttt{'deleted'}.
    \item \texttt{email\_verified\_at} (timestamptz, null): Thời điểm xác thực email.
    \item \texttt{created\_at}, \texttt{updated\_at} (timestamptz): Thời gian tạo và cập nhật.
\end{itemize} \\ \hline

\textbf{profiles} \newline (hồ\_sơ) &
\begin{itemize}
    \item \texttt{id} (text, khóa chính).
    \item \texttt{user\_id} (FK → \texttt{users.id}): Liên kết tài khoản người dùng.
    \item \texttt{full\_name} (text, null): Họ và tên.
    \item \texttt{phone\_number} (text, null): Số điện thoại.
    \item \texttt{avatar\_url} (text, null): Đường dẫn ảnh đại diện.
    \item \texttt{created\_at}, \texttt{updated\_at} (timestamptz): Thời gian tạo và cập nhật.
\end{itemize} \\ \hline

\textbf{roles} \newline (vai\_trò) &
\begin{itemize}
    \item \texttt{id} (text, khóa chính).
    \item \texttt{code} (text, duy nhất): Mã vai trò — \texttt{'MANAGER'}, \texttt{'TRAINER'}, \texttt{'STAFF'}, \texttt{'MEMBER'}.
    \item \texttt{name} (text): Tên hiển thị.
\end{itemize} \\ \hline

\textbf{user\_role \_assignments} \newline (phân\_vai\_trò) &
\begin{itemize}
    \item \texttt{id} (text, khóa chính).
    \item \texttt{user\_id} (FK → \texttt{users.id}): Người dùng được gán.
    \item \texttt{role\_id} (FK → \texttt{roles.id}): Vai trò được gán.
    \item \texttt{branch\_id} (FK → \texttt{branches.id}, null): Chi nhánh áp dụng; null nếu phạm vi toàn hệ thống.
    \item \texttt{assigned\_at} (timestamptz): Thời điểm phân công.
\end{itemize} \\ \hline

\section{Mối quan hệ giữa các thực thể}

\subsection*{Nhóm người dùng và phân quyền}
\begin{itemize}
    \item \textbf{users – profiles}: Một–một (1:1). Mỗi tài khoản có tối đa một hồ sơ cá nhân.
    \item \textbf{users – user\_role\_assignments}: Một–nhiều (1:N). Một người dùng có thể được gán nhiều vai trò ở các chi nhánh khác nhau.
    \item \textbf{roles – user\_role\_assignments}: Một–nhiều (1:N). Một vai trò có thể được gán cho nhiều người dùng.
\end{itemize}

\subsection*{Nhóm tổ chức và chi nhánh}
\begin{itemize}
    \item \textbf{organizations – branches}: Một–nhiều (1:N). Một tổ chức có nhiều chi nhánh.
    \item \textbf{branches – branch\_manager\_assignments}: Một–nhiều (1:N). Một chi nhánh có nhiều đợt phân công quản lý theo thời gian.
    \item \textbf{users – branch\_manager\_assignments}: Một–nhiều (1:N). Một người dùng có thể quản lý nhiều chi nhánh ở các thời điểm khác nhau.
\end{itemize}

\subsection*{Nhóm gói tập hội viên (theo thời gian)}
\begin{itemize}
    \item \textbf{users – subscriptions}: Một–nhiều (1:N). Một thành viên có thể có nhiều đăng ký kế tiếp nhau.
    \item \textbf{membership\_plans – subscriptions}: Một–nhiều (1:N). Một gói tập có thể được nhiều thành viên đăng ký.
    \item \textbf{subscriptions – subscription\_status\_history}: Một–nhiều (1:N). Mỗi đăng ký lưu lịch sử các lần đổi trạng thái.
    \item \textbf{subscriptions – member\_check\_ins}: Một–nhiều (1:N). Mỗi đăng ký có nhiều lần check-in thực tế.
\end{itemize}

\subsection*{Nhóm gói tập theo buổi (khóa học)}
\begin{itemize}
    \item \textbf{course\_packages – course\_enrollments}: Một–nhiều (1:N). Một gói (30 buổi, 60 buổi) có thể được nhiều học viên đăng ký.
    \item \textbf{users – course\_enrollments}: Một–nhiều (1:N). Một học viên có thể đăng ký nhiều khóa học (ràng buộc ứng dụng: chỉ một đăng ký \texttt{ACTIVE} tại một thời điểm).
    \item \textbf{course\_enrollments – class\_attendance}: Một–nhiều (1:N). Mỗi lượt điểm danh làm giảm số buổi còn lại của đăng ký.
\end{itemize}

\subsection*{Nhóm ca tập và phân công huấn luyện viên}
\begin{itemize}
    \item \textbf{branches – shifts}: Một–nhiều (1:N). Mỗi chi nhánh có nhiều ca tập theo lịch.
    \item \textbf{shifts – trainer\_assignments}: Một–nhiều (1:N, tối đa 3). Một ca tập có tối đa 3 HLV được phân công; ràng buộc duy nhất ngăn phân công trùng.
    \item \textbf{users (HLV) – trainer\_assignments}: Một–nhiều (1:N). Một HLV có thể được phân công nhiều ca.
    \item \textbf{shifts – class\_attendance}: Một–nhiều (1:N). Một ca tập có nhiều bản ghi điểm danh học viên.
    \item \textbf{shifts – member\_body\_measurements}: Một–nhiều (1:N). Ca tập được liên kết để khóa số đo sau khi ca kết thúc.
\end{itemize}

\subsection*{Nhóm tập thử và chuyển đổi}
\begin{itemize}
    \item \textbf{branches – trial\_bookings}: Một–nhiều (1:N). Mỗi chi nhánh có nhiều lịch tập thử.

\subsection*{Tóm tắt các loại quan hệ chính}
\begin{itemize}
    \item \textbf{Một–một (1:1)}: \texttt{users – profiles}, \texttt{users – staff\_profiles}, \texttt{users – member\_health\_profiles}.
    \item \textbf{Một–nhiều (1:N)}: hầu hết các quan hệ còn lại (chi nhánh – ca tập, người dùng – đăng ký, hóa đơn – thanh toán, tài sản – phiếu bảo trì, \ldots).
    \item \textbf{Nhiều–nhiều (N:N)} qua bảng trung gian:
        \begin{itemize}
            \item \texttt{users} (HLV) – \texttt{shifts} thông qua \texttt{trainer\_assignments}.
            \item \texttt{users} – \texttt{roles} thông qua \texttt{user\_role\_assignments}.
        \end{itemize}
\end{itemize}
